${student.firstName.charAt(0)}${student.lastName.charAt(0)}

Performance Chart

}); }); // Animate chart bars setTimeout(() => { document.querySelectorAll('.chart-bar').forEach(bar => { bar.style.opacity = '1'; }); }, 100); } // Calculate quarter grade function calculateQuarterGrade(grades) { if (!grades || grades.writtenWorks === '-' || grades.performanceTasks === '-' || grades.quarterlyAssessment === '-') { return 0; } // Weights: Written Works (40%), Performance Tasks (40%), Quarterly Assessment (20%) return (grades.writtenWorks * 0.4) + (grades.performanceTasks * 0.4) + (grades.quarterlyAssessment * 0.2); } // Get color based on grade function getGradeColor(grade) { if (grade === '-' || grade === 'N/A') return 'text-gray-500'; document.getElementById('description').value = student.description || ''; // Set the current quarter to 1 for editing currentQuarter = 1; quarterTabs.forEach(tab => { tab.classList.remove('active'); if (tab.dataset.quarter === '1') { tab.classList.add('active'); `; } // Open grade edit modal function openGradeEditModal(studentId, quarter) { const student = students.find(s => s.id === studentId); if (!student) return; document.getElementById('gradeModalTitle').textContent = `Edit Quarter ${quarter} Grades`; document.getElementById('gradeStudentId').value = studentId; document.getElementById('gradeQuarter').value = quarter; const grades = student.grades && student.grades[quarter] ? parseInt(writtenWorksInput.value) : '-'; const performanceTasks = performanceTasksInput.value ? parseInt(performanceTasksInput.value) : '-'; const quarterlyAssessment = quarterlyAssessmentInput.value ? parseInt(quarterlyAssessmentInput.value) : '-'; grades[quarter] = { writtenWorks, performanceTasks, quarterlyAssessment }; } } const student = { firstName, lastName, section, studentId, profilePicture, description, grades, updatedAt: firebase.firestore.FieldValue.serverTimestamp() }; try { let docRef; if (isEditing && id) { // Update existing student docRef = db.collection('students').doc(id); await docRef.update(student); // Update local array const index = students.findIndex(s => s.id === id); if (index !== -1) { students[index] = { ...student, id }; } showToast('Student updated successfully!'); } else { // Add new student student.createdAt = firebase.firestore.FieldValue.serverTimestamp(); docRef = await db.collection('students').add(student); // Add to local array students.push({ ...student, id: docRef.id }); showToast('Student added successfully!'); } hideModal(studentModal); loadStudents(); // If this was the selected student, update the details view if (selectedStudentId === (id || docRef.id)) { const updatedStudent = students.find(s => s.id === (id || docRef.id)); renderStudentDetails(updatedStudent); } // Log event to analytics analytics.logEvent(isEditing ? 'student_updated' : 'student_added', { section: section }); } catch (error) { console.error('Error saving student:', error); showToast('Failed to save student. Please try again.', 'error'); } finally { saveBtn.disabled = false; spinner.classList.add('hidden'); } } // Save grades to Firebase async function saveGrades() { const saveBtn = document.getElementById('saveGradeBtn'); const spinner = document.getElementById('gradeSpinner'); saveBtn.disabled = true; spinner.classList.remove('hidden'); const studentId = document.getElementById('gradeStudentId').value; const quarter = document.getElementById('gradeQuarter').value; const writtenWorks = document.getElementById('editWrittenWorks').value ? parseInt(document.getElementById('editWrittenWorks').value) : '-'; const performanceTasks = document.getElementById('editPerformanceTasks').value ? parseInt(document.getElementById('editPerformanceTasks').value) : '-'; const quarterlyAssessment = document.getElementById('editQuarterlyAssessment').value ? parseInt(document.getElementById('editQuarterlyAssessment').value) : '-'; try { const studentRef = db.collection('students').doc(studentId); // Update the specific quarter grades using dot notation const updateData = {}; updateData[`grades.${quarter}`] = { writtenWorks, performanceTasks, quarterlyAssessment }; updateData.updatedAt = firebase.firestore.FieldValue.serverTimestamp(); await studentRef.update(updateData); // Update local array const student = students.find(s => s.id === studentId); if (student) { if (!student.grades) { student.grades = {}; } student.grades[quarter] = { writtenWorks, performanceTasks, quarterlyAssessment }; } hideModal(gradeEditModal); loadStudents(); // If this was the selected student, update the details view if (selectedStudentId === studentId) { renderStudentDetails(student); } showToast('Grades updated successfully!'); // Log event to analytics analytics.logEvent('grades_updated', { quarter: quarter }); } catch (error) { console.error('Error saving grades:', error); showToast('Failed to save grades. Please try again.', 'error'); } finally { saveBtn.disabled = false; spinner.classList.add('hidden'); } } // Delete student from Firebase async function deleteStudent(id) { if (confirm('Are you sure you want to delete this student? This action cannot be undone.')) { showLoading('Deleting student...'); try { await db.collection('students').doc(id).delete(); // Remove from local array students = students.filter(student => student.id !== id); loadStudents(); // Clear details if the deleted student was selected if (selectedStudentId === id) { selectedStudentId = null; studentDetails.innerHTML = `

Select a student to view details

Student information and grades will appear here

`; } showToast('Student deleted successfully!'); // Log event to analytics analytics.logEvent('student_deleted'); } catch (error) { console.error('Error deleting student:', error); showToast('Failed to delete student. Please try again.', 'error'); } finally { hideLoading(); } } } // Export data function exportData() { const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(students, null, 2)); const downloadAnchorNode = document.createElement('a'); downloadAnchorNode.setAttribute("href", dataStr); downloadAnchorNode.setAttribute("download", "emman_edutrack_data.json"); document.body.appendChild(downloadAnchorNode); downloadAnchorNode.click(); downloadAnchorNode.remove(); // Log event to analytics analytics.logEvent('data_exported'); } // Show modal function showModal(modal) { modal.classList.remove('hidden'); setTimeout(() => { modal.style.opacity = '1'; }, 10); } // Hide modal function hideModal(modal) { modal.style.opacity = '0'; setTimeout(() => { modal.classList.add('hidden'); }, 300); }
